• 1.postcss & autoprefixer

    源文件src

    
            .test{
              display:flex;
            }

    编译后dist

    
            .test{
              display: -webkit-box;
              display: -webkit-flex;
              display: -ms-flexbox;
              display: flex;
            }

  • 2.postcss-pxtorem & posthtml-px2rem

    px->rem 需开启选项

    
            .test{
              display:flex;
              padding:60px;
              margin-top:30px;
              font-size:32px;
              line-height:1.2;
              border:1px solid #dfdfdf;
            }

    注：计算比例75：1；支持style方式和内联方式

    编译后dist

    
            .test{
              display:flex;
              padding:.8rem;
              margin-top:.4rem;
              font-size:.426rem;
              line-height:1.2;
              border:1px solid #dfdfdf; //1px 不进行转换 可设置minPixelValue
            }

  • 3.http-proxy-middleware

    需配置请求api地址

    
            {
              "proxy":"http://test.api.com", //api接口地址
            }

  • 4.gulp-preprocess

    需配置各环境api请求地址，具体环境变量（stg,pre,prd）可自行修改

    
            
            {
               // @if NODE_ENV = 'development'
               env: 'dev',
               // @endif
               // @if NODE_ENV = 'test'
               env: 'stg',
               // @endif
               // @if NODE_ENV = 'prelease'
               env: 'rep',
               // @endif
               // @if NODE_ENV = 'production'
               env: 'prd',
               // @endif
            }
            
            {
              "scripts": {
                "dev": "gulp dev",
                "test": "set NODE_ENV=test&&gulp build",
                "pre": "set NODE_ENV=prelease&&gulp build",
                "prd": "set NODE_ENV=production&&gulp build"
                }
            }
